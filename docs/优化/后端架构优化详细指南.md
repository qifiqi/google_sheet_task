### 架构优势
1. **模块化设计**：基本按功能划分了不同模块
2. **数据库优化**：已实现连接池、索引优化、监控功能
3. **日志系统**：完善的日志记录和管理
4. **配置管理**：动态配置系统，支持实时更新
5. **异步任务**：支持后台任务执行和调度

### 架构劣势
1. **单文件过大**：多个文件超过500行，违反单一职责原则
2. **职责混乱**：路由层包含业务逻辑，服务层职责不清
3. **紧耦合**：模块间依赖关系复杂，难以独立测试
4. **缺乏抽象**：没有统一的基类和接口定义
5. **错误处理分散**：异常处理逻辑散布在各个文件中

### 2. 基础类和接口设计

#### 2.1 基础实体类
```python
# app/domain/entities/base.py
from abc import ABC
from datetime import datetime
from typing import Any, Dict, Optional
from uuid import uuid4

class BaseEntity(ABC):
    """基础实体类"""
    
    def __init__(self, id: Optional[str] = None):
        self.id = id or str(uuid4())
        self.created_at = datetime.utcnow()
        self.updated_at = datetime.utcnow()
    
    def update_timestamp(self):
        """更新时间戳"""
        self.updated_at = datetime.utcnow()
    
    def to_dict(self) -> Dict[str, Any]:
        """转换为字典"""
        return {
            'id': self.id,
            'created_at': self.created_at.isoformat(),
            'updated_at': self.updated_at.isoformat()
        }
    
    def __eq__(self, other):
        if not isinstance(other, BaseEntity):
            return False
        return self.id == other.id
    
    def __hash__(self):
        return hash(self.id)
```

#### 2.3 统一响应格式
```python
# app/core/utils/response.py
from typing import Any, Optional, Dict
from datetime import datetime
from flask import jsonify
from enum import Enum

class ResponseStatus(Enum):
    SUCCESS = "success"
    ERROR = "error"
    WARNING = "warning"

class APIResponse:
    """统一API响应格式"""
    
    @staticmethod
    def success(
        data: Any = None, 
        message: str = "操作成功",
        meta: Optional[Dict] = None
    ) -> tuple:
        """成功响应"""
        response = {
            "status": ResponseStatus.SUCCESS.value,
            "message": message,
            "data": data,
            "timestamp": datetime.utcnow().isoformat()
        }
        
        if meta:
            response["meta"] = meta
            
        return jsonify(response), 200
    
    @staticmethod
    def error(
        message: str,
        code: int = 400,
        details: Any = None,
        error_code: Optional[str] = None
    ) -> tuple:
        """错误响应"""
        response = {
            "status": ResponseStatus.ERROR.value,
            "message": message,
            "timestamp": datetime.utcnow().isoformat()
        }
        
        if details:
            response["details"] = details
            
        if error_code:
            response["error_code"] = error_code
            
        return jsonify(response), code
    
    @staticmethod
    def paginated(
        data: list,
        page: int,
        per_page: int,
        total: int,
        message: str = "获取成功"
    ) -> tuple:
        """分页响应"""
        total_pages = (total + per_page - 1) // per_page
        
        meta = {
            "pagination": {
                "page": page,
                "per_page": per_page,
                "total": total,
                "total_pages": total_pages,
                "has_next": page < total_pages,
                "has_prev": page > 1
            }
        }
        
        return APIResponse.success(data=data, message=message, meta=meta)
```

from app.core.utils.response import APIResponse
from app.application.commands.task_commands import CreateTaskCommand, UpdateTaskCommand

tasks_bp = Blueprint('tasks', __name__)

@tasks_bp.route('', methods=['GET'])
def list_tasks():
    """获取任务列表"""
    try:
        query = ListTasksQuery(
            page=request.args.get('page', 1, type=int),

        return APIResponse.paginated(**result)
    except Exception as e:

@tasks_bp.route('', methods=['POST'])
def create_task():
    """创建任务"""
    try:
        command = CreateTaskCommand(request.get_json())
        result = command.execute()
        return APIResponse.success(data=result, message="任务创建成功")
    except ValidationError as e:
        return APIResponse.error(f"数据验证失败: {str(e)}", 422)
    except Exception as e:
        return APIResponse.error(f"创建任务失败: {str(e)}")

from typing import Dict, Any, List, Optional
from app.domain.entities.task import Task
from app.domain.repositories.task import TaskRepository
from app.domain.services.task_domain import TaskDomainService
from app.infrastructure.external.notifications.dingtalk import DingTalkNotifier

class TaskApplicationService:

        # 发送通知
        await self._notifier.notify_task_created(created_task)
        
        return created_task
    
    async def start_task(self, task_id: str) -> bool:
        """启动任务"""
        task = await self._task_repository.get_by_id(task_id)
        if not task:
            raise TaskNotFoundError(task_id)
        
        # 检查任务是否可以启动
        if not self._task_domain_service.can_start_task(task):
            raise TaskCannotStartError(f"任务 {task_id} 当前状态不允许启动")
        
        # 检查并发限制
        running_count = await self._task_repository.count(status='running')
        max_concurrent = self._get_max_concurrent_tasks()
        
        if running_count >= max_concurrent:
            raise ConcurrencyLimitExceededError(
                f"已达到最大并发任务数 {max_concurrent}"
            )
        
        # 启动任务
        task.start()
        await self._task_repository.update(task)
        
        # 异步执行任务
        await self._execute_task_async(task)
        
        return True

        ).first()

        """获取所有实体"""
        query = self._session.query(self._model_class)
        query = self._apply_filters(query, filters)
        
        models = query.all()
        return [self._model_to_entity(model) for model in models]

        model = self._entity_to_model(entity)
        self._session.add(model)
        self._session.commit()
        self._session.refresh(model)
        
        return self._model_to_entity(model)

        """更新实体"""
        model = self._session.query(self._model_class).filter(
            self._model_class.id == entity.id
        ).first()
        
        if not model:
            raise EntityNotFoundError(f"实体 {entity.id} 不存在")
        
        self._update_model_from_entity(model, entity)
        self._session.commit()
        self._session.refresh(model)
        
        return self._model_to_entity(model)
    
    async def delete(self, id: str) -> bool:
        """删除实体"""
        model = self._session.query(self._model_class).filter(
            self._model_class.id == id
        ).first()
        
        if not model:
            return False
        
        self._session.delete(model)
        self._session.commit()
        return True

        """检查实体是否存在"""
        count = self._session.query(self._model_class).filter(
            self._model_class.id == id
        ).count()
        
        return count > 0

                    query = query.filter(getattr(self._model_class, key).in_(value))
                else:
                    query = query.filter(getattr(self._model_class, key) == value)
        
        return query

        """模型转实体"""

                print(f"警告: 无法加载secrets文件 {self.secrets_file}: {e}")

        """获取敏感信息"""

# app/infrastructure/cache/redis_cache.py
import json
import pickle
from typing import Any, Optional, Union
from redis import Redis
from app.core.config import get_config

class RedisCache:
    """Redis缓存客户端"""
    
    def __init__(self, redis_client: Optional[Redis] = None):
        self.redis = redis_client or Redis.from_url(get_config().get_redis_uri())
    
    def get(self, key: str, default: Any = None) -> Any:
        """获取缓存值"""
        try:
            value = self.redis.get(key)
            if value is None:
                return default
            return pickle.loads(value)
        except Exception as e:
            logger.error(f"缓存获取失败: {e}")
            return default
    
    def set(self, key: str, value: Any, ttl: int = 3600) -> bool:
        """设置缓存值"""
        try:
            serialized_value = pickle.dumps(value)
            return self.redis.setex(key, ttl, serialized_value)
        except Exception as e:
            logger.error(f"缓存设置失败: {e}")
            return False
    
    def delete(self, key: str) -> bool:
        """删除缓存"""
        try:
            return bool(self.redis.delete(key))
        except Exception as e:
            logger.error(f"缓存删除失败: {e}")
            return False
    
    def exists(self, key: str) -> bool:
        """检查缓存是否存在"""
        try:
            return bool(self.redis.exists(key))
        except Exception as e:
            logger.error(f"缓存检查失败: {e}")
            return False
    
    def clear_pattern(self, pattern: str) -> int:
        """清除匹配模式的缓存"""
        try:
            keys = self.redis.keys(pattern)
            if keys:
                return self.redis.delete(*keys)
            return 0
        except Exception as e:
            logger.error(f"批量缓存清除失败: {e}")
            return 0

# 缓存装饰器
def cached(ttl: int = 3600, key_prefix: str = ""):
    """缓存装饰器"""
    def decorator(func):
        def wrapper(*args, **kwargs):
            # 生成缓存键
            cache_key = f"{key_prefix}:{func.__name__}:{hash(str(args) + str(kwargs))}"
            
            # 尝试从缓存获取
            cached_result = cache.get(cache_key)
            if cached_result is not None:
                return cached_result
            
            # 执行函数并缓存结果
            result = func(*args, **kwargs)
            cache.set(cache_key, result, ttl)
            return result
        
        return wrapper
    return decorator

# 全局缓存实例
