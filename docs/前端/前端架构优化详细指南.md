### 当前技术栈
- **模板引擎**: Jinja2
- **CSS框架**: Bootstrap 5.1.3 (CDN)
- **图标库**: Bootstrap Icons + Font Awesome
- **JavaScript**: 原生JavaScript (内联代码)
- **AJAX**: XMLHttpRequest (原生)
- **构建工具**: 无

### 架构优势
1. **响应式设计**: 使用Bootstrap实现基本响应式布局
2. **模板继承**: 合理使用Jinja2模板继承机制
3. **实时更新**: 实现了基本的轮询机制更新数据
4. **用户友好**: 基本的加载状态和错误提示

### 架构劣势
1. **文件过大**: 多个模板文件超过500行，严重违反单一职责原则
2. **代码重复**: 大量重复的JavaScript和CSS代码
3. **内联代码**: JavaScript和CSS直接写在HTML中，难以维护
4. **缺乏模块化**: 没有JavaScript模块化设计
5. **性能问题**: 重复加载相同资源，缺乏缓存策略
6. **可维护性差**: 代码组织混乱，难以定位和修改

## JavaScript模块化改造

### 1. 核心API客户端

```javascript
// static/js/core/api.js
class APIClient {
    constructor(baseURL = '/api', options = {}) {
        this.baseURL = baseURL;
        this.defaultOptions = {
            headers: {
                'Content-Type': 'application/json',
                ...options.headers
            },
            credentials: 'same-origin',
            ...options
        };
    }

    async request(endpoint, options = {}) {
        const url = `${this.baseURL}${endpoint}`;
        const config = {
            ...this.defaultOptions,
            ...options,
            headers: {
                ...this.defaultOptions.headers,
                ...options.headers
            }
        };

        try {
            const response = await fetch(url, config);
            
            if (!response.ok) {
                throw new APIError(
                    `HTTP ${response.status}: ${response.statusText}`,
                    response.status,
                    await response.json().catch(() => null)
                );
            }

            const contentType = response.headers.get('content-type');
            if (contentType && contentType.includes('application/json')) {
                return await response.json();
            }
            
            return await response.text();
        } catch (error) {
            if (error instanceof APIError) {
                throw error;
            }
            throw new APIError(`网络请求失败: ${error.message}`);
        }
    }

    // GET请求
    async get(endpoint, params = {}) {
        const url = new URL(endpoint, this.baseURL);
        Object.keys(params).forEach(key => {
            if (params[key] !== undefined && params[key] !== null) {
                url.searchParams.append(key, params[key]);
            }
        });
        
        return this.request(url.pathname + url.search);
    }

    // POST请求
    async post(endpoint, data = null) {
        return this.request(endpoint, {
            method: 'POST',
            body: data ? JSON.stringify(data) : null
        });
    }

    // PUT请求
    async put(endpoint, data = null) {
        return this.request(endpoint, {
            method: 'PUT',
            body: data ? JSON.stringify(data) : null
        });
    }

    // DELETE请求
    async delete(endpoint) {
        return this.request(endpoint, {
            method: 'DELETE'
        });
    }
}

// 自定义API错误类
class APIError extends Error {
    constructor(message, status = 0, details = null) {
        super(message);
        this.name = 'APIError';
        this.status = status;
        this.details = details;
    }
}

// 导出单例实例
const apiClient = new APIClient();
export { APIClient, APIError, apiClient };
```

// static/js/components/BaseComponent.js
class BaseComponent {
    constructor(element, options = {}) {
        this.element = typeof element === 'string' 
            ? document.querySelector(element) 
            : element;
            
        if (!this.element) {
            throw new Error(`Element not found: ${element}`);
        }

        this.options = { ...this.defaultOptions, ...options };
        this.state = {};
        this.events = new Map();
        
        this.init();
    }

    // 默认选项，子类可以重写
    get defaultOptions() {
        return {};
    }

    // 初始化方法，子类必须实现
    init() {
        throw new Error('init() method must be implemented by subclass');
    }

    // 事件绑定
    on(event, handler) {
        if (!this.events.has(event)) {
            this.events.set(event, []);
        }
        this.events.get(event).push(handler);
        return this;
    }

    // 触发事件
    emit(event, data = null) {
        if (this.events.has(event)) {
            this.events.get(event).forEach(handler => {
                handler.call(this, data);
            });
        }
        return this;
    }

    // 移除事件监听
    off(event, handler = null) {
        if (!this.events.has(event)) return this;
        
        if (handler) {
            const handlers = this.events.get(event);
            const index = handlers.indexOf(handler);
            if (index > -1) {
                handlers.splice(index, 1);
            }
        } else {
            this.events.delete(event);
        }
        return this;
    }

    // 状态管理
    setState(newState) {
        const oldState = { ...this.state };
        this.state = { ...this.state, ...newState };
        this.onStateChange(this.state, oldState);
        return this;
    }

    // 状态变化回调，子类可以重写
    onStateChange(newState, oldState) {
        // 默认实现为空
    }

    // DOM操作辅助方法
    $(selector) {
        return this.element.querySelector(selector);
    }

    $$(selector) {
        return Array.from(this.element.querySelectorAll(selector));
    }

    // 显示/隐藏元素
    show() {
        this.element.style.display = '';
        return this;
    }

    hide() {
        this.element.style.display = 'none';
        return this;
    }

    // 添加/移除CSS类
    addClass(className) {
        this.element.classList.add(className);
        return this;
    }

    removeClass(className) {
        this.element.classList.remove(className);
        return this;
    }

    toggleClass(className) {
        this.element.classList.toggle(className);
        return this;
    }

    // 销毁组件
    destroy() {
        this.events.clear();
        this.element = null;
        this.options = null;
        this.state = null;
    }
}

export default BaseComponent;

### 4. 任务卡片组件

```javascript
// static/js/components/TaskCard.js
import BaseComponent from './BaseComponent.js';
import TaskService from '../services/TaskService.js';

class TaskCard extends BaseComponent {
    get defaultOptions() {
        return {
            showActions: true,
            showProgress: true,
            autoRefresh: false,
            refreshInterval: 5000
        };
    }

    init() {
        this.taskId = this.element.dataset.taskId;
        this.refreshTimer = null;
        
        this.bindEvents();
        
        if (this.options.autoRefresh) {
            this.startAutoRefresh();
        }
    }

    bindEvents() {
        // 绑定操作按钮事件
        const startBtn = this.$('.btn-start');
        const stopBtn = this.$('.btn-stop');
        const deleteBtn = this.$('.btn-delete');

        if (startBtn) {
            startBtn.addEventListener('click', () => this.startTask());
        }
        
        if (stopBtn) {
            stopBtn.addEventListener('click', () => this.stopTask());
        }
        
        if (deleteBtn) {
            deleteBtn.addEventListener('click', () => this.deleteTask());
        }
    }

    async startTask() {
        try {
            this.setLoading(true);
            await TaskService.startTask(this.taskId);
            this.emit('task:started', { taskId: this.taskId });
            await this.refresh();
        } catch (error) {
            this.emit('task:error', { error, action: 'start' });
        } finally {
            this.setLoading(false);
        }
    }

    async stopTask() {
        try {
            this.setLoading(true);
            await TaskService.stopTask(this.taskId);
            this.emit('task:stopped', { taskId: this.taskId });
            await this.refresh();
        } catch (error) {
            this.emit('task:error', { error, action: 'stop' });
        } finally {
            this.setLoading(false);
        }
    }

    async deleteTask() {
        if (!confirm('确定要删除这个任务吗？')) return;
        
        try {
            this.setLoading(true);
            await TaskService.deleteTask(this.taskId);
            this.emit('task:deleted', { taskId: this.taskId });
            this.element.remove();
        } catch (error) {
            this.emit('task:error', { error, action: 'delete' });
            this.setLoading(false);
        }
    }

    async refresh() {
        try {
            const task = await TaskService.getTask(this.taskId);
            this.updateTaskData(task.data);
        } catch (error) {
            this.emit('task:error', { error, action: 'refresh' });
        }
    }

    updateTaskData(taskData) {
        // 更新任务名称
        const nameEl = this.$('.task-name');
        if (nameEl) nameEl.textContent = taskData.name;

        // 更新任务状态
        const statusEl = this.$('.task-status');
        if (statusEl) {
            statusEl.textContent = taskData.status;
            statusEl.className = `task-status badge badge-${this.getStatusClass(taskData.status)}`;
        }

        // 更新进度条
        if (this.options.showProgress) {
            this.updateProgress(taskData.progress || 0);
        }

        // 更新按钮状态
        this.updateActionButtons(taskData.status);
    }

    updateProgress(progress) {
        const progressBar = this.$('.progress-bar');
        if (progressBar) {
            progressBar.style.width = `${progress}%`;
            progressBar.textContent = `${progress}%`;
        }
    }

    updateActionButtons(status) {
        const startBtn = this.$('.btn-start');
        const stopBtn = this.$('.btn-stop');
        
        if (startBtn) {
            startBtn.disabled = !['pending', 'error', 'completed'].includes(status);
        }
        
        if (stopBtn) {
            stopBtn.disabled = status !== 'running';
        }
    }

    getStatusClass(status) {
        const statusMap = {
            'pending': 'secondary',
            'running': 'primary',
            'completed': 'success',
            'error': 'danger',
            'cancelled': 'warning'
        };
        return statusMap[status] || 'secondary';
    }

    setLoading(loading) {
        const buttons = this.$$('button');
        buttons.forEach(btn => {
            btn.disabled = loading;
        });

        if (loading) {
            this.addClass('loading');
        } else {
            this.removeClass('loading');
        }
    }

    startAutoRefresh() {
        this.refreshTimer = setInterval(() => {
            this.refresh();
        }, this.options.refreshInterval);
    }

    stopAutoRefresh() {
        if (this.refreshTimer) {
            clearInterval(this.refreshTimer);
            this.refreshTimer = null;
        }
    }

    destroy() {
        this.stopAutoRefresh();
        super.destroy();
    }
}

export default TaskCard;
```
